[docs]class IkResults(object):
    """
    Class with hold result of a inverse kinematics query
    """

    def __init__(self, path, error_codes):
        self.__path = path
        self.__error_codes = [ActionResult(x.val) for x in error_codes]

    @property
    def path(self):
        """
        path : JointPath
            solutions from the inverse kinematics
        """
        return self.__path

    @property
    def error_codes(self):
        """
        error_code : List[ActionResult]
            error codes
        """
        return self.__error_codes

    @property
    def succeeded(self):
        if all([e == ActionResult.SUCCESS
                for e in self.__error_codes]):
            return True
        else:
            return False

    def __str__(self):
        return str(self.__error_codes)

    def __repr__(self):
        return self.__str__()
